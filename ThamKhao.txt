from flask import Flask, jsonify
import redis
import os
import socket

app = Flask(__name__)

# Kết nối đến cụm Redis lưu trữ ngữ cảnh chung
REDIS_HOST = os.getenv("REDIS_HOST", "10.100.219.196")
r = redis.Redis(host=REDIS_HOST, port=6379, db=0, decode_responses=True)

@app.route('/video_stream')
def video_stream():
    # Đọc định danh của Pod hiện tại
    pod_name = socket.gethostname()
    
    # Đọc số khung hình hiện tại từ Redis, nếu chưa có thì khởi tạo là 0
    current_frame = r.get("ue1_session")
    frame_number = int(current_frame) if current_frame else 0
    
    # Tăng frame (mô phỏng tiến trình UE đọc luồng dữ liệu liên tục)
    frame_number += 1
    r.set("ue1_session", frame_number)
    
    return jsonify({
        "serving_pod": pod_name,
        "current_frame": frame_number,
        "status": "active"
    })

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8000)

cat <<EOF > ~/drl-agent-build/agent.py
import time, os, urllib.request

URL = "http://localhost:8080/metrics"

def get_rsrp():
    try:
        with urllib.request.urlopen(URL, timeout=2) as res:
            data = res.read().decode()
            for line in data.split('\n'):
                # Tìm dòng chứa giá trị RSRP cho ue1
                if 'ueransim_ue_rsrp{pod_name="ue1"}' in line:
                    return float(line.split()[-1])
    except Exception as e:
        return None
    return None

def update_mec_node(node_name):
    print(f"🔄 ĐANG HANDOVER sang Node: {node_name}")
    patch_cmd = f'kubectl patch deployment mec-counter-app -n open5gs -p \'{{"spec":{{"template":{{"spec":{{"nodeSelector":{{"kubernetes.io/hostname":"{node_name}"}}}}}}}}}}\''
    os.system(patch_cmd)

current_node = "open5gs"
print("📡 Agent đã sẵn sàng theo dõi...")

while True:
    rsrp = get_rsrp()
    if rsrp is None:
        print("❌ Chưa nhận được dữ liệu từ Exporter (port 8080)...")
    else:
        print(f"📡 RSRP thực: {rsrp:.1f} dBm | Node hiện tại: {current_node}")
        
        # Logic Handover (Vùng trễ từ -80 đến -90)
        if current_node == "open5gs" and rsrp < -90.0:
            update_mec_node("workerk8s")
            current_node = "workerk8s"
        elif current_node == "workerk8s" and rsrp > -80.0:
            update_mec_node("open5gs")
            current_node = "open5gs"
            
    time.sleep(1)
